    1. bitxor：

    用位运算实现异或操作：

    算了接下来直接贴源码吧。

    * bitXor - x^y using only ~ and & 
 *   Example: bitXor(4, 5) = 1
 *   Legal ops: ~ &
 *   Max ops: 14
 *   Rating: 1
 */
int bitXor(int x, int y) {
//  return 2;
    return ~(x&y)&~(~x&~y);   //不是同为0的情况和不是同为1的情况下进行位与操作
}

    2.tmin:

    * tmin - return minimum two's complement integer 
 *   Legal ops: ! ~ & ^ | + << >>
 *   Max ops: 4
 *   Rating: 1
 */
int tmin(void) {
  //return 2;
  return 1<<31;   //int是32位，补码最小值是符号位为1，然后剩余31都是0，故这里通过把1左移31位得到这个值
}

    3.isTmax:

    * isTmax - returns 1 if x is the maximum, two's complement number,
 *     and 0 otherwise 
 //通过位运算计算是否是补码最大值
 *   Legal ops: ! ~ & ^ | +
 *   Max ops: 10
 *   Rating: 1
 */
int isTmax(int x) {
  //return 2;
  //对于int来说，补码最大值是符号位为0，然后剩余31位为1
  //！符号表示取反，对于整型，不为0的取反后是0,0取反后是1
  //~符号则是对每一位(bit)取反
  int e1=!(~x); //如果x=-1,-1的补码是全1（符号位为1），即1111..11，所以~(-1)将得到全0，然后！0得到 1
  //以上是排除x=-1的情形
  //若x不等于-1的话，即其他情况最后都将得到 e1=0
  return !((~(x+1)^x)|e1);
  //若x是Tmax最大值01111..11，+1后将变为最小值1000..00，~进行各位取反，1000..00又重新变为0111..11
  //然后和自己本身异或，得0（这种~(x+1)^x得0的情况只出现于Tmax和-1两种情形，而之前我们已经排除了x=-1的情况，
  //所以现在还成立的话就说明x是Tmax）
  //接下去，e1已经是0了，然后 0|0=0 ，最后!0=1
  //所以最后逻辑就是，若x是Tmax即补码最大值，则return 1（这也是题目要求的），否则return 0
}

    4.allOddBits:

    * allOddBits - return 1 if all odd-numbered bits in word set to 1
 *   where bits are numbered from 0 (least significant) to 31 (most significant)
 *   Examples allOddBits(0xFFFFFFFD) = 0, allOddBits(0xAAAAAAAA) = 1
 *   Legal ops: ! ~ & ^ | + << >>
 *   Max ops: 12
 *   Rating: 2
 */
int allOddBits(int x) {
  //return 2;
  //判断所有奇数位是否都为1，这里的奇数指的是位的阶级是2的奇数次幂
  //第一步先是构造掩码
  int mask=0xAA+(0xAA<<8);
  //int是4字节，0xAA<<8得到 1010 1010 0000 0000，加上0xAA，最终得到 1010 1010 1010 1010
  //我在想直接赋值 int mask=0xAAAA 不行吗？
  mask=mask+(mask<<16); //哎呦我去，这个有什么意义吗，左移16位是全0，mask加上之后还是等于mask本身啊
  return !((mask&x)^mask); //如果一个数和掩码mask与之后还能等于mask，就说明这个数的奇数位上是1
  //与mask相等，异或结果为0，取反为1，根据题目要求return 1
}

    5.negate:

    * negate - return -x 
 *   Example: negate(1) = -1.
 *   Legal ops: ! ~ & ^ | + << >>
 *   Max ops: 5
 *   Rating: 2
 */
int negate(int x) {
  //得到某个数的负数，不能直接使用 "-"操作符
  return ~x+1; //套用结论：各位取反，末尾+1
}

    6.isAsciiDigit:

    * isAsciiDigit - return 1 if 0x30 <= x <= 0x39 (ASCII codes for characters '0' to '9')
 *   Example: isAsciiDigit(0x35) = 1.
 *            isAsciiDigit(0x3a) = 0.
 *            isAsciiDigit(0x05) = 0.
 *   Legal ops: ! ~ & ^ | + << >>
 *   Max ops: 15
 *   Rating: 3
 */
int isAsciiDigit(int x) {
  //x处于0x30~0x39之间就返回1
  int sign=1<<31; //将得到 100000..00 ，符号位为1
  //加上比0x39大的数后符号由正变负:
  int big=~(sign|0x39); //构造上界，如果超过，符号位为1 
  //加上比0x30小的值时是负数:
  int small=(~(0x30)+1);  //构造下界，如果低于，则符号位为1
  //我还是不明白它这里为什么要这样构造上下界？
  big=sign&(big+x)>>31;
  small=sign&(small+x)>>31;
  //最终结果都是要>>31的，所以就是看符号位了
  //如果符号位为1（超过或低于），则&sign之后依然为1，反之为0
  return !(big|small);//超过了big的话big的结果为1，同理低于small的话small的结果为1，只要两者有一个为1，"|"的结果就为1，！1=0
  //只有处在中间才都为0，然后！0=1，return 1

}

    7.conditional:

    * conditional - same as x ? y : z 
 *   Example: conditional(2,4,5) = 4
 *   Legal ops: ! ~ & ^ | + << >>
 *   Max ops: 16
 *   Rating: 3
 */
int conditional(int x, int y, int z) {
  //使用位级运算实现C语言中的 x?y:z三目运算符
  //若x不等于0，则取y的值，反之取z的值
  //前提知识：任何不等于0的数值经过“！”取反之后都为0，而再次取反则归为1
  x=!!x; //通过两次取反操作即可把不同的数值变换为只有两种结果：
  //若原先数值不等于0，则!!x之后变为1（即0000 0001） //为了搞清楚我写了C程序进行验证，得到确实为0000 0001
  //若原先数值等于0,则!!x之后变为0（全0，即0000 0000）
  x=~x+1; //求0或1的补码，0的补码是其本身，位表示全0（0000 0000）；1的补码是-1，位表示全1（1111 1111）
  return (x&y)|(~x&z);
  //若x不等于0，则!!x之后变为1，然后变为它的补码-1即1111 1111 ，~x则为 0000 0000,所以无论z取什么&之后终将为0，结果就是取y的值
  //若x等于0，则!!x之后变为0，然后变为它的补码0000 0000 ，~x则为1111 1111,所以无论y取什么&之后终将为0，结果就是取z的值
  //终于搞清楚了。
}

    8.isLessOrEqual:

    * isLessOrEqual - if x <= y  then return 1, else return 0 
 *   Example: isLessOrEqual(4,5) = 1.
 *   Legal ops: ! ~ & ^ | + << >>
 *   Max ops: 24
 *   Rating: 3
 */
int isLessOrEqual(int x, int y) {
  int fux=~x+1;  //得到-x
  int cha=y+fux; // y-x
  int sign=(cha>>31)&1;  
  //若x>y则y-x必定会产借位操作，sign就是负数，符号位就一定为1
  //右移31位然后&1之后就落实了sign的符号位为1
  //若y>=x则sign为0
  int xsign=(x>>31)&1; //取出x的符号位
  int ysign=(y>>31)&1;//取出y的符号位
  int bitxor=xsign^ysign;  //判断x和y的符号位是否一致，相同为0
  return (bitxor&xsign)|((!bitxor)&(!sign));
  //要么x和y符号相同且x<=y， !0 & !0 =1
  //要么x和y符号不同且x<0
  //噢终于搞明白了。
}

    9.logicalNeg:

    * logicalNeg - implement the ! operator, using all of 
 *              the legal operators except !
 *   Examples: logicalNeg(3) = 0, logicalNeg(0) = 1
 *   Legal ops: ~ & ^ | + << >>
 *   Max ops: 12
 *   Rating: 4 
 */
int logicalNeg(int x) {
  return ((x|(~x+1))>>31)+1;
  //x按位取反，末尾+1得到-x，然后进行x|-x
  //如果x是0的话，x|-x之后不影响，符号位仍为0
  //如果x不为0的话，x|-x之后因为负数的影响，符号位必定为1
  //右移31位取其符号位
  //0的补码还是0，所以+1=1
  //1的补码是-1.所以+1=0
}

    10.howManyBits:

    * howManyBits - return the minimum number of bits required to represent x in
 *             two's complement
 *  Examples: howManyBits(12) = 5
 *            howManyBits(298) = 10
 *            howManyBits(-5) = 4
 *            howManyBits(0)  = 1
 *            howManyBits(-1) = 1
 *            howManyBits(0x80000000) = 32
 *  Legal ops: ! ~ & ^ | + << >>
 *  Max ops: 90
 *  Rating: 4
 */
int howManyBits(int x) {
  //题目：求值“一个数用补码表示最少需要几位？”
  //原理：对于正数，从高位到低位，找第一个位是1的（比如是n），再加上符号位，所以最少需要n+1位
  //对于负数，从高位到低位，找第一个位是0的（比如是n），再加上符号位，所以最少是n位
  //通俗来讲：
  //如果是一个正数，则需要找到它最高的一位（假设是n）是1的，再加上符号位，结果为n+1；
  //如果是一个负数，则需要知道其最高的一位是0的（例如4位的1101和三位的101补码表示的是一个值：-3，最少需要3位来表示）。
  int b16,b8,b4,b2,b1,b0; //表示0~15,16~23,24~27,28~29，30,31的位置处是否含有1，如有，则对其赋值需要的位数
  int sign=x>>31; //取符号位
  x=(sign&~x)|(~sign&x); //如x为正数则不变，x为负数则取反，这里是为了统一正负数，我们之后只用找到含有1的位置即可

  b16=!!(x>>16)<<4;//先看低16位是否含有1，若有则表示至少需要16位，所以给b16赋值为16 (1<<4=16，秒啊)
  x=x >> b16;//若低16位有1，则原数右移16位，因为上一句已经确定是否至少需要16位(针对0~15)，若没有1，则b16=0，原数就不需要位移，本句也就无视掉就好了，继续往下执行

  b8=!!(x>>8)<<3; //看剩余位的高8位是否含有1，若有则表示至少还需要8位，给b8赋为8 (1<<3=8)
  x=x>>b8; //同理

  b4=!!(x>>4)<<2;
  x=x>>b4;

  b2=!!(x>>2)<<1;
  x=x>>b2;

  b1=!!(x>>1);
  x=x>>b1;

  b0=x;

  return b16+b8+b4+b2+b1+b0+1;   //最后加上符号位
  //不明觉厉呜呜
}
