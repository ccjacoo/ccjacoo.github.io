    学习：

    最流行的局域网技术是以太网（Ethernet）。一个以太网段（Ethernet segment）包括一些电缆（通常是双绞线）和集线器。一端连接到主机的适配器，然后另一端连到集线器。集线器不加分辨地把一个端口上收到的每个位复制到其他所有端口上。一台主机可以发送一段位（帧）到这个网段内的其他任何主机上。帧包括header（标识源和目的地址以及帧的长度），以及数据位的有效载荷。每个主机适配器都能看到这个帧，但是只有目的主机才能真正读取它。

    使用一些电缆和网桥能够把多个以太网段连接成较大的局域网，叫做桥接以太网（bridged Ethernet）。

    多个不兼容的局域网可以用路由器（router，是一种特殊的计算机）连接起来。每台路由器对于它所连接到的每个网络都有一个适配器（端口）。

    一个IP地址就是一个32位无符号整数。结构如下：

    /* IP address structure*/
struct in_addr{
    uint32_t s_ddr;     //Adress in network byte order is big-endian .
}

    客户端和服务端使用 socket 函数来创建一个套接字描述符 （socket descriptor）。

    #include <sys/types.h>
#include <sys/socket.h>

int socket(int domain, int type, int protocol); //若成功则返回非负描述符，若失败返回-1

//也可以直接硬编码：（此时返回的clientfd还是不能用于读写的）
clientfd = socket(AF_INET, SOCK_STREAM, 0);
/*
AF_INET 表示我们正在使用32位IP地址
SOCK_STREAM 表示这个套接字是连接的一个端点
*/

    客户端调用 connect 还是来建立和服务器的连接：

    #include <sys/socket.h>
int connect(int clientfd, const struct sockaddr *addr, socklen_t addrlen);
//成功返回0，失败返回-1

    connect函数试图与套接字地址为addr的服务器建议因特网连接。其中 addrlen是sizeof(sockaddr_in)

    如果连接成功，clientfd描述符现在就准备好可以用于读写了。

    服务器通过 bind、listen、accept 函数来和客户端建立连接。

    #include <sys/socket.h>

int bind(int sockfd, const struct sockaddr *addr, socklen_t addrlen);
//成功返回0，失败返回-1

int listen(int sockfd, int backlog);
//成功返回0，失败返回-1

int accept(int listenfd, struct sockaddr *addr, int *addrlen);
//成功返回0，失败返回-1

    bind函数告诉内核将 addr中的服务器套接字地址 与 套接字描述符 sockfd 联系（bind绑定）起来。

    listen函数将 sockfd 从一个 主动套接字 转化为一个 监听套接字（listening socket），因此该套接字可以接收来自客户端的连接请求。backlog一般设为较大的值，比如1024.

    accept函数接收来自客户端的连接请求给监听描述符listenfd。addr填写的是客户端的套接字地址，并返回一个已连接描述符（connected descripor）。

    监听描述符（listening descripor）与已连接描述符（connected descriptor）的区分：监听描述符一般只需创建一次，然后存在于服务器的整个生命周期； 而已连接描述符则只是客户端与服务端之间建立的某一次连接的端点，服务器每次接受连接请求时都会被创建一次，它存在于服务器为一个客户端服务的过程中。

    大致过程：服务器调用accept函数等待到达监听描述符的连接请求。客户端调用connect函数发送一个连接请求给服务器端的listenfd。服务器端利用accept函数打开了一个新的已连接描述符connfd。因此就在clientfd和connfd之间建立了连接，并且随后返回connfd给应用程序。此时客户端和服务端就可以分别通过读和写 clientfd 和 connfd 来回传送数据了。

    Linux提供了 getaddrinfo 函数将主机名、主机地址、服务名和端口号的字符串转化为套接字结构的地址。

    #include <sys/types.h>
#include <sys/socket.h>
#include <netdb.h>

int getaddrinfo(const char *host, const char *service, 
                const struct addrinfo *hints, struct addrinfo **result);
//成功返回0，失败返回非0的错误代码

void freeaddrinfo(struct addrinfo *result);   //return 0；

const char *gai_strerror(int errcode);   //return wrong data

    只给定host和service即可，getaddrinfo便可以返回result，result是一个指向addrinfo结构的链表（更像是套接字数据库）。其中每个结构指向一个套接字地址。

    在客户端调用 getaddrinfo函数后，会遍历链表，依次尝试每一个套接字地址，直到调用socket和connect成功然后建立起连接。同样的，服务器也会尝试遍历 链表中的每个套接字地址，直到调用socket和bind成功，然后描述符会被绑定到一个合法的套接字地址。

    getnameinfo函数则是反过来，将一个套接字结构地址转换为响应的主机和服务名字符串，并将它们复制到host和service缓冲区。

    #include <sys/socket.h>
#include <netdb.h>

int getnameinfo(const struct sockaddr *sa, socklen_t salen,
               char *host, size_t hostlen, char *service, 
               size_t servlen, int flags)
/*
sa 指向大小为salen字节的套接字地址结构
host 指向大小为hostlen字节的缓冲区
service指向大小为servlen字节的缓冲区
*/

    如果getnameinfo返回非0的错误代码，则应用程序可以调用 gai_strerror把它转化为字符串。

    echo服务端：

    #include "csapp.h"

void echo(int connfd);

int main(int argc, char **argv)
{
    int listenfd,connfd;
    socklen_t clientlen;
    struct sockaddr_storage clientaddr;  // enough space for any address
    char client_hostname[MAXLINE], client_port[MAXLINE];

    if(argc!=2){
        fprintf(stderr, "usage: %s <host> <port>\n",argv[0]);
        exit(0);
    }

    listenfd= Open_listenfd(argv[1]);
    while(1){
        clientlen=sizeof(struct sockaddr_storage);
        connfd=Accept(listenfd, (SA *)&clientaddr, &clientlen);
        Getnameinfo((SA *)&clientaddr,clientlen,client_hostname,MAXLINE,
            client_port,MAXLINE,0);
        printf("Connected to (%s , %s)\n",client_hostname,client_port);
        echo(connfd);
        Close(connfd);
    }
    exit(0);
}

    这里的服务器一次只能处理一个客户端，一次一个地在客户端间迭代，称为 迭代服务器（iterative server）。

    关于EOF：在连接中，当一个进程关闭它的那一端时会产生EOF，而连接的另一端的进程在试图读取stream中最后一个字节之后的数据时会检测到EOF。

    web服务器以两种不同的方式向客户端提供内容：

    1. 取一个磁盘文件，并将它的内容返回给客户端。磁盘文件被称为静态文件（static content）。
    2. 运行一个可执行文件，并将它的输出返回给客户端。运行时产生的输出为动态内容（dynamic content）

    一般区分是否静态or动态的做法是：把全部静态文件放到同一的目录中，把动态可执行的文件放到另一个目录中。加以区分。

A simple ,iterative HTTP/1.0 Web server that 
uses the GET method to serve static and dynamic content.
*/

void doit(int fd);

int parse_uri(char *uri, char *filename, char *cgiargs);

void get_filetype(char *filename, char *filetype);
void serve_dynamic(iint fd, char *cause,
    char *errnum, char *shortmsg, char *longmsg);

int int main(int argc, char const **argv)
{
    int listenfd,connfd;
    char hostname[MAXLINE],port[MAXLINE];
    socklen_t clientlen;
    struct sockaddr_storage clientaddr;

    /*Check command-line args*/
    if(args!=2){
        fprintf(stderr, "usage: %s <port>\n",argv[0]);
        exit(0);
    }

    listenfd=Open_listenfd(argv[1]);
    while(1){
        clientlen=sizeof(clientaddr);
        connfd=Accept(listenfd, (SA *)&clientaddr, &clientlen);
        Getnameinfo((SA *)&clientaddr,clientlen,hostname,MAXLINE,
            port,MAXLINE,0);
        printf("Connected to (%s , %s)\n",hostname,port);
        doit(connfd);
        Close(connfd);
    }
}


/* 处理一个HTTP事务*/
void doit(int fd){
    int is_static;
    struct stat_sbuf;
    char buf[MAXLINE], method[MAXLINE], uri[MAXLINE], version[MAXLINE];
    char filename[MAXLINE], cgiargs[MAXLINE];
    rio_t rio;

    /*read request line and headers*/
    Rio_readinitb(&rio,fd);
    Rio_readlineb(&rio,buf,MAXLINE);
    printf("Request headers: \n");
    printf("%s", buf);
    sscanf(buf,"%s %s %s",method,uri,version);
    if(strcasecmp(method,"GET")){
        clienterror(fd,method,"501","Not implemented",
            "Tiny does not implement this method");
        return;
    }
    read_requsthdrs(&rio);

    /*parse URI from GET request*/
    is_static=parse_uri(uri,filename,cgiargs);
    if(stat(filename,&sbuf)<0){
        clienterror(fd,filename,"404","Not Found",
            "Tiny couldn't find this file");
        return;
    }

    if(is_static){ //serve static content
        if(!(S_ISREG(sbuf.st_mode)) || !(S_IRUSR & sbuf.st_mode)){
            clienterror(fd, filename,"403","Forbidden!",
                "Tiny couldn't read the file");
            return;
        }
        serve_static(fd,filename,sbuf.st_size);
    }
    else{  //serve dynamic content
        if(!(S_ISREG(sbuf.st_mode)) || !(S_IXUSR & sbuf.st_mode)){
            clienterror(fd, filename,"403","Forbidden!",
                "Tiny couldn't run the CGI program");
            return;
        }
        serve_dynamic(fd,filename,cgiargs);
    }
}

    char buf[MAXLINE],body[MAXLINE];

    /*Build the HTTP response body*/
    sprintf(body,"<html><title>Tiny Error</title>");
    sprintf(body,"%s<body bgcolor=""ffffff"">\r\n",body);
    sprintf(body,"%s%s:%s\r\n",body,errnum,shortmsg);
    sprintf(body,"%s<p>%s: %s\r\n",body,longmsg,cause);
    sprintf(body,"%s<hr><em>The Tiny Web server</em>\r\n",body);

    /*Print the HTTP response*/
    sprintf(buf,"HTTP/1.0 %s %s\r\n",errnum,shortmsg);
    Rio_writen(fd,buf,strlen(buf));
    sprintf(buf,"Content-type: text/html\r\n");
    Rio_writen(fd,buf,strlen(buf));
    sprintf(buf,"Content-length: %d\r\n\r\n",(int)strlen(body));
    Rio_writen(fd,buf,strlen(buf));
    Rio_writen(fd,body,strlen(body));
}


/*不使用请求报头中的任何信息，用该函数读取并忽略这些报头*/
void read_requsthdrs(rio_t *rp){
    char buf[MAXLINE];

    Rio_readlineb(rp,buf,MAXLINE);
    while(strcmp(buf, "\r\n")){ //终止请求报头的空文本行是由回车和换行符组成的，check this
        Rio_readlineb(rp,buf,MAXLINE);
        printf("%s",buf);
    }
    return;
}

//假设静态内容的主目录就是它的当前牡蛎，而可执行文件的主目录是 ./cgi-bin
//包含任何字符串 cgi-bin 的URI都会被认为是对动态内容的请求
//默认的文件名是 ./home.html
int parse_uri(char *uri,char *filename,char *cgiargs){
    char *ptr;

    if(!strstr(uri,"cgi-bin")){ //static content
        strcpy(cgiargs,""); //如果请求的是静态内容，则清除CGI参数字符串
        strcpy(filename,"."); //将URI转换为一个Linux相对路径名
        strcat(filename,uri);
        if(uri[strlen(uri)-1]=='/'){  //如果URI是用'/'结尾的
            strcat(filename,"home.html"); //把默认的文件名加在其后面
        }
        return 1;
    }
    else{   //dynamic content
        ptr=index(uri,'?');   //抽取出所有的CGI参数
        if(ptr){
            strcpy(cgiargs,ptr+1);
            *ptr='\0';
        }
        else{
            strcpy(cgiargs,"");
        }
        strcpy(filename,"."); //URI剩余部分转换为一个Linux相对文件名
        strcat(filename,uri);
        return 0;
    }
}
