    ucore lab6：

    练习1：分析 Round Robin 调度算法：

    Round Robin 调度算法的思想是让所有runnable态的进程分时轮流使用CPU的时间，调度器维护一个有序运行队列，当前进程的时间片用完（time_silce==0）之后,调度器将当前进程放置到运行队列的尾部，再从其头部取出进程进行调度。

    这个算法实现在 default_sched.c 中。

    //进程队列的初始化
staic void RR_init(struct run_queue *rq){
    list_init(&(rq->run_list));
    rq->proc_num=0;
}

//旧进程返回rq队列
staic void RR_enqueue(struct run_queue *rq, struct proc_struct *proc){
    assert(list_empty(&(proc->run_link))); //assert函数返回false则终止程序并打印错误信息
    //把进程控制块指针(proc->run_link)放到rq队列(rq->run_list)的末尾
    list_add_before(&(rq->run_list), &(proc->run_link));
    //进程控制块的时间片(proc->time_slice)为0
    if(proc->time_slice==0 || proc->time_slice > rq->max_time_slice){
        proc->time_slice=rq->max_time_slice; //重置，下次还可以再被执行
    }
    proc->rq=rq;
    rq->proc_num ++; //rq的进程数目+1
}

//新进程出列
staic void RR_dequeue(struct run_queue *rq, struct proc_struct *proc){
    assert(!list_empty(&(proc->run_link)) && proc->rq==rq);
    //把就绪进程队列rq的进程控制块指针的队列元素删除
    list_del_init(&(proc->run_link));
    rq->proc_num --;  //就绪进程个数-1
}

staic struct proc_struct * RR_pick_next(struct run_queue *rq){
    //选取就绪进程队列rq中的队头队列元素
    list_entry_t *le=list_next(&(rq->run_list));  //队头进程
    //置为当前占用CPU的进程
    if(le!=&(rq->run_list)){
        return le2proc(le,run_link); //把队列元素转换成进程控制块指针
    }
    return NULL;
}

staic void RR_proc_tick(struct run_queue *rq, struct proc_struct *proc){
    if(proc->time_slice > 0){
        proc->time_slice --;
    }
    if(proc->time_slice == 0){
        proc->need_resched = 1;
        //下次中断执行trap函数时，由于进程成员变量need_resched为1
        //因而执行schedule函数，把当前进程放回就绪队列末尾
        //然后取出就绪队列上等待时间最久的进程(队首)
    }
}

//对象化，类的实现，提供调度算法的切换接口
struct sched_class default_sched_class = {
    .name="RR_scheduler",
    .init=RR_init,
    .enqueue=RR_enqueue,
    .dequeue=RR_dequeue,
    .pick_next=RR_pick_next,
    .proc_tick=RR_proc_tick,
};

    练习2：实现 Stride Scheduling 调度算法：

    Stride Scheduling调度算法实现在 default_sched_stride.c 中，并且要覆盖原先的算法Round Robin即 default_sched.c 。

    Stride Scheduling调度算法思想： （分配优先级）

    1. 为每个 runnable 态的进程设置一个当前状态stride，表示该进程当前的调度权。另外还要定义对应的pass值，表示该进程在调度后，stride需要进行的累加值！

    2. 每次需要调度的时候，从当前所有runnable的进程中选择 stride最小的进程进行调度。对于获得调度的某进程P，将对应的stride加上其对应的pass值（只与进程的优先权有关）。

    3. 在一段时间后回到步骤2，重新调度当前stride最小的进程。

    /*完整代码*/

#include <defs.h>
#include <list.h>
#include <proc.h>
#include <assert.h>
#include <default_sched.h>

#define USE_SKEW_HEAP 1
//you should define the BigStride constant here 
#define BIG_STRIDE 0x7FFFFFFF //定义一个大整数处以优先级

//the compare function for two skew_heap_node_t and the corresponding procs
//优先队列的比较函数
staic int proc_stride_comp_f(void *a, void *b){
    struct proc_struct *p =le2proc(a, lab6_run_pool);
    struct proc_struct *q =le2proc(b, lab6_run_pool);
    int32_t c = p->lab6_stride - q->lab6_stride; //步数相减，确定大小关系
    if(c>0) return 1;
    else if(c==0) return 0;
    else return -1;
}

//运行队列初始化
staic void stride_init(struct run_queue *rq){
    list_init(&(rq->run_list)); //初始化调度器类的信息
    rq->lab6_run_pool = NULL;  //初始化当前的运行队列为一个空的容器结构
    rq->proc_num = 0;  //初始化进程数为0
}

//入队
staic void stride_enqueue(struct run_queue *rq, struct proc_struct *proc){
#if USE_SKEW_HEAP
    rq->lab6_run_pool=skew_heap_insert(rq->lab6_run_pool, &(proc->lab6_run_pool), proc_stride_comp_f)
    //比较队头元素与当前进程的步数大小，选择步数小的运行
#else
    assert(list_empty(&(proc->run_link)));
    list_add_before(&(rq->run_list), &(proc->run_link));
    //将proc插入运行队列中
#endif
    if(proc->time_slice==0 || proc->time_slice > rq->max_time_slice){
        proc->time_slice = rq->max_time_slice; //重置
    }
    proc->rq=rq;
    rq->proc_num ++;
}

//出队
staic void stride_dequeue(struct run_queue *rq, struct proc_struct *proc){
#if USE_SKEW_HEAP
    rq->lab6_run_pool=skew_heap_remove(rq->lab6_run_pool, &(proc->lab6_run_pool, proc_stride_comp_f))
#else 
    assert(!list_empty(&(proc->run_link)) && proc->rq==rq);
    list_del_init(&(proc->run_link));
#endif
    rq->proc_num --;
}

static struct proc_struct *stride_pick_next(struct run_queue *rq) {
#if USE_SKEW_HEAP
     if (rq->lab6_run_pool == NULL) return NULL;
     struct proc_struct *p = le2proc(rq->lab6_run_pool, lab6_run_pool);
#else
     list_entry_t *le = list_next(&(rq->run_list)); //这里的le还只是头结点？
     if (le == &rq->run_list)
        return NULL;

     struct proc_struct *p = le2proc(le, run_link);
     le = list_next(le);
     while (le != &rq->run_list)
     {

        struct proc_struct *q = le2proc(le, run_link);
        if ((int32_t)(p->lab6_stride - q->lab6_stride) > 0)
            p = q; //P永远表示stride值最小的那个进程
        le = list_next(le);
     }
#endif
     if (p->lab6_priority == 0)
        p->lab6_stride += BIG_STRIDE;  //更新stride值
     else p->lab6_stride += BIG_STRIDE / p->lab6_priority;
     return p; //返回当前stride最小的进程
}

//时间片函数 -- 检测当前进程是否已用完分配的时间片
static void stride_proc_tick(struct run_queue *rq, struct proc_struct *proc) {
     if (proc->time_slice > 0) {
          proc->time_slice --;
     }
     if (proc->time_slice == 0) {
          proc->need_resched = 1;  //设置标识
     }
}

//封装类
struct sched_class default_sched_class = {
     .name = "stride_scheduler",
     .init = stride_init,
     .enqueue = stride_enqueue,
     .dequeue = stride_dequeue,
     .pick_next = stride_pick_next,
     .proc_tick = stride_proc_tick,
};
